## 7. Methodology

```python
from sklearn.metrics import confusion_matrix
import numpy as np

def fairness_audit(y_true, y_pred, protected_attr):
    groups = np.unique(protected_attr)
    g0 = protected_attr == groups[0]
    g1 = protected_attr == groups[1]

    # Demographic Parity
    dp_diff = abs(y_pred[g0].mean() - y_pred[g1].mean())

    # Equal Opportunity (TPR parity)
    tpr0 = y_pred[(g0) & (y_true==1)].mean()
    tpr1 = y_pred[(g1) & (y_true==1)].mean()
    eo_diff = abs(tpr0 - tpr1)

    # Disparate Impact
    di = min(y_pred[g0].mean(), y_pred[g1].mean()) / \
         max(y_pred[g0].mean(), y_pred[g1].mean())

    return {"DP_diff": dp_diff, "EO_diff": eo_diff, "DI": di}
```
